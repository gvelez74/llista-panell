💳 Crear taula llista_finances — Supabase
Executa aquest SQL a Supabase → SQL Editor per crear la taula de finances. La connexió s'usa la mateixa que el mòdul Seguiment.
CREATE TABLE llista_finances (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id TEXT DEFAULT 'llista-cat',
  type TEXT NOT NULL DEFAULT 'expense',
  is_budget BOOLEAN DEFAULT false,
  categoria TEXT DEFAULT 'ALTRES',
  concept TEXT DEFAULT '',
  description TEXT DEFAULT '',
  amount DECIMAL(10,2) DEFAULT 0,
  month INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE llista_finances ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON llista_finances
  FOR ALL USING (true) WITH CHECK (true);

CREATE TABLE llista_tasks (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  title TEXT NOT NULL,
  description TEXT DEFAULT '',
  status TEXT DEFAULT 'pendent',
  prioritat TEXT DEFAULT 'normal',
  assignat TEXT DEFAULT '',
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE llista_tasks ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON llista_tasks
  FOR ALL USING (true) WITH CHECK (true);

CREATE TABLE llista_members (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  name TEXT NOT NULL,
  color TEXT DEFAULT '#6366F1',
  created_at TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE llista_members ENABLE ROW LEVEL SECURITY;

  FOR ALL USING (true) WITH CHECK (true);
